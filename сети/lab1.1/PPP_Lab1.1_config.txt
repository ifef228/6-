!    DTE — там, где "DTE V.35 TX and RX clocks detected"
! 2. На DCE задайте clock rate, затем на обоих — encapsulation ppp и IP
! 3. Проверка: ping между маршрутизаторами, с ПК — ping и WEB
!

! ######################## МАРШРУТИЗАТОР DCE ########################
! Подключите этот конфиг к маршрутизатору со стороны DCE (show controllers serial 0/0/0)

hostname RouterDCE_67_19
no ip domain-lookup

! Последовательный интерфейс (PPP) — сторона DCE: clock rate обязателен
interface Serial0/0/0
 encapsulation ppp
 ip address 176.67.19.1 255.255.255.252
 clock rate 250000
 no shutdown

! LAN со стороны DCE (сеть с WEB-сервером 126.67.19.10)
interface GigabitEthernet0/0
 ip address 126.67.19.1 255.255.255.0
 no shutdown

! Статический маршрут до сети DTE (195.67.19.0) через DTE
ip route 195.67.19.0 255.255.255.0 176.67.19.2

end
write memory

! ######################## МАРШРУТИЗАТОР DTE ########################
! Подключите этот конфиг к маршрутизатору со стороной DTE

hostname RouterDTE_67_19
no ip domain-lookup

! Последовательный интерфейс (PPP) — сторона DTE (clock rate НЕ ставим)
interface Serial0/0/0
 encapsulation ppp
 ip address 176.67.19.2 255.255.255.252
 no shutdown

! LAN со стороны DTE (локальная сеть 195.67.19.n). Шлюз для ПК = 195.67.19.5
interface GigabitEthernet0/0
 ip address 195.67.19.5 255.255.255.0
 no shutdown

! Статический маршрут до сети с WEB-сервером (126.67.19.0) через DCE
ip route 126.67.19.0 255.255.255.0 176.67.19.1

end
write memory

! ============================================================
! ПРОВЕРКА (группа 67, вариант 19)
! 1. На маршрутизаторе DTE: ping 176.67.19.1
! 2. На маршрутизаторе DCE: ping 176.67.19.2
! 3. Проверка PPP: show interfaces serial 0/0/0
!    Должно быть: Encapsulation PPP, line protocol is up (connected)
! 4. С ПК в сети 195.67.19.0: ping 126.67.19.10 и браузер http://www.Фамилия_ИО.iu5
! ============================================================
! WEB-сервер: IP 126.67.19.10, маска 255.255.255.0, шлюз 126.67.19.1
! DNS (сеть DTE): запись www.Фамилия_ИО.iu5 -> 126.67.19.10
! ПК в сети DTE: IP 195.67.19.n, шлюз 195.67.19.5, DNS — IP DNS-сервера
! ============================================================

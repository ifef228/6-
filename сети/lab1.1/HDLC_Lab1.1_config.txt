! ######################## МАРШРУТИЗАТОР DCE ########################
! Подключите этот конфиг к маршрутизатору со стороны DCE (show controllers serial 0/0/0)

hostname RouterDCE_67_19
no ip domain-lookup

! Последовательный интерфейс (HDLC) — сторона DCE: clock rate обязателен
interface Serial0/0/0
 encapsulation hdlc
 ip address 176.67.19.1 255.255.255.252
 clock rate 250000
 no shutdown

! LAN со стороны DCE (сеть с WEB-сервером 126.67.19.10)
interface GigabitEthernet0/0
 ip address 126.67.19.1 255.255.255.0
 no shutdown

! Статический маршрут до сети DTE (195.67.19.0) через DTE
ip route 195.67.19.0 255.255.255.0 176.67.19.2

end
write memory

! ######################## МАРШРУТИЗАТОР DTE ########################
! Подключите этот конфиг к маршрутизатору со стороны DTE

hostname RouterDTE_67_19
no ip domain-lookup

! Последовательный интерфейс (HDLC) — сторона DTE (clock rate НЕ ставим)
interface Serial0/0/0
 encapsulation hdlc
 ip address 176.67.19.2 255.255.255.252
 no shutdown

! LAN со стороны DTE (локальная сеть 195.67.19.n). Шлюз для ПК = 195.67.19.5
interface GigabitEthernet0/0
 ip address 195.67.19.5 255.255.255.0
 no shutdown

! Статический маршрут до сети с WEB-сервером (126.67.19.0) через DCE
ip route 126.67.19.0 255.255.255.0 176.67.19.1

end
write memory

! На маршрутизаторе DCE: ping 176.67.19.2

! ============================================================
! WEB-сервер: IP 126.67.19.10, маска 255.255.255.0, шлюз 126.67.19.1
